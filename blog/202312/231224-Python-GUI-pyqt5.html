Python 图形界面GUI开发库 PYQT5 快速入门

本期内容介绍如何使用 Qt designer来辅助制作Python程序的界面。与使用tkinter时，需要通过代码的方式来管理GUI不同，Qt designer 提供了界面编辑器，可以通过所见即所得的方式来开发UI，非常方便。

PyQt5有两种授权方式，非开源商用需要购买授权，免费使用针对的是开源项目。 如果要正式的商用可以选择PySide2。这两个开发库非常的相似，文档基本都是可以共用的。

安装pyqt5 

pip install pyqt5

商用可以使用 PySide2

动作确认

import sys
from PyQt5 import QtWidgets

def show_window():
    app = QtWidgets.QApplication(sys.argv)
    window = QtWidgets.QWidget()
    label1 = QtWidgets.QLabel(window)
    label1.setText("Qt UI")
    window.setWindowTitle('PyQt5 Window')
    window.setGeometry(200, 50, 400, 250)
    label1.move(10,10)
    window.show()
    sys.exit(app.exec_())

if __name__ == '__main__':
    show_window()

    然后在 qt5_applications -> Qt -> bin 下找到 designer.exe 文件。
    在文件上点右键，创建其桌面的快捷链接。

在设计界面时，就可以通过此链接来打开 Qt designer 了。

  • 准备程序，导入UI文件，并显示。

推荐用法：将uic导入的UI设置为主类的界面

import sys
from PyQt5 import QtWidgets, uic

class MainUIApp(QtWidgets.QMainWindow):
    def __init__(self):
        super(MainUIApp, self).__init__()
        # 加载界面
        self.ui = uic.loadUi("MainUI.ui", self)

        # 连接信号和槽
        self.ui.someButton.clicked.connect(self.handleButtonClick)

    def handleButtonClick(self):
        # 处理按钮点击事件
        print("Button has been clicked")

if __name__ == "__main__":
    app = QtWidgets.QApplication(sys.argv)
    mainWindow = MainUIApp()
    mainWindow.show()
    sys.exit(app.exec_())

  10. 多线程

    使用QTimer
    特点: 定时执行一个函数，比如用于制作读秒器，或者定时的更新数据。 优点是代码很简单。

    fomr PyQt5 import QtCore
timer = QtCore.QTimer()
timer.timeout.connect(displayImage) # 指定要运行的函数
timer.start(1000) # 时间间隔为1s

    使用QThread
    特点：可以实现复杂的多线程程序。可以在类中实现何时来调用函数，从而处理更加复杂的任务。

    class WorkThread(QThread):
    timer = pyqtSignal() # 自定义信号，进行通讯
    def run(self):
        while True:
            self.sleep(1)
            self.timer.emit() # 发送通讯
    
worker = WorkThread()
worker.timer.connect(displayImage) # 绑定执行函数
worker.start()

  11. 视频显示
    下载视频：https://samplelib.com/sample-mp4.html
    下面是一个在QLabel控件中播放视频的代码。（播放速度控制有些问题）

    from PyQt5.QtCore import QThread, pyqtSignal
from PyQt5.QtGui import QImage, QPixmap
import cv2
from PyQt5 import QtGui

class WorkThreadVideo(QThread):
    change_pixmap_signal = pyqtSignal(QImage)  # Signal to indicate a frame is ready to display
    def __init__(self, video_source="demo.MP4"):
        super().__init__()
        # Open the video source
        self.vid = cv2.VideoCapture(video_source)
        if not self.vid.isOpened():
            raise ValueError("Unable to open video source", video_source)
        # Get video source width and height
        self.width = self.vid.get(cv2.CAP_PROP_FRAME_WIDTH)
        self.height = self.vid.get(cv2.CAP_PROP_FRAME_HEIGHT)

    def run(self):
        while self.vid.isOpened():
            loop_start_time = time.time()  # 记录循环开始的时间
            ret, frame = self.vid.read()
            if ret:
                # Convert the frame to QT format
                rgb_image = cv2.cvtColor(frame, cv2.COLOR_BGR2RGB)
                h, w, ch = rgb_image.shape
                bytes_per_line = ch * w
                convert_to_Qt_format = QImage(rgb_image.data, w, h, bytes_per_line, QImage.Format_RGB888)
                p = convert_to_Qt_format.scaled(360, 280, QtCore.Qt.KeepAspectRatio, QtCore.Qt.SmoothTransformation)
                self.change_pixmap_signal.emit(p)  # Emit the ready frame
            else:
                return
            
            #cv2.imshow("win",frame)
            #if cv2.waitKey(1) & 0xFF == ord('q'):
            #    break
            
        # Release the video source when the loop is finished
        self.vid.release()
        return

def displayVideo(img):
    dlg1.label_Video.setPixmap(QtGui.QPixmap.fromImage(img))
    
# 设置Lable容器Size
dlg1.label_Video.setFixedWidth(200)
dlg1.label_Video.setFixedHeight(400)
