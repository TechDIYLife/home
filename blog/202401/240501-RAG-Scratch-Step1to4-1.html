import bs4
from langchain import hub
from langchain.text_splitter import RecursiveCharacterTextSplitter
from langchain_community.document_loaders import WebBaseLoader
from langchain_community.vectorstores import Chroma
from langchain_core.output_parsers import StrOutputParser
from langchain_core.runnables import RunnablePassthrough
from langchain_community.llms import Ollama
from langchain_community.embeddings import OllamaEmbeddings
from langchain.prompts import ChatPromptTemplate

# Load Documents
url="https://techdiylife.github.io/blog/202401/240327-ollama-20question.html"
loader = WebBaseLoader(web_paths=(url,),)
docs = loader.load()

# Split
text_splitter = RecursiveCharacterTextSplitter(chunk_size=300, chunk_overlap=50)
splits = text_splitter.split_documents(docs)

# Embed (终端运行此命令下载模型：ollama pull mofanke/acge_text_embedding)
vectorstore = Chroma.from_documents(documents=splits, embedding=OllamaEmbeddings(model="mofanke/acge_text_embedding"))

retriever = vectorstore.as_retriever()
print("indexing done")

# 使用自定义prompt
template = """你是一个回答问题的助手。请使用以下检索到的背景信息来回答问题。如果你不知道答案，直接说你不知道。请用最多三句话来保持回答的简洁性。
问题：{question} 

背景：{context} 

答案：
"""
prompt = ChatPromptTemplate.from_template(template)
print(prompt)

llm = Ollama(model="qwen:72b")
# Chain
rag_chain = (
    {"context": retriever | format_docs, "question": RunnablePassthrough()}
    | prompt
    | llm
    | StrOutputParser()
)

# Question
rag_chain.invoke("ollama支持哪些模型?")

# RAG测试集
dataset=["ollama支持哪些模型?",
            "Linux系统上，模型下载后存放到哪里了？",
            "windows下如何修改模型默认下载目录？",
            "Modelfile是做什么用的？",
            "如何导入safetensors格式的模型？",
            "有什么webui可用吗？",
            "Windows如何卸载ollama？",
            "Linux如何卸载ollama？",
            "如何让ollma只使用第二张GPU？",
            "Ollama支持RAG吗？"]
for idx, _q in enumerate(dataset, start=1):
    print(f"问题{idx}：{_q}")
    print(rag_chain.invoke(_q))

Q7： RAG是如何进行检索的？

参考Part2部分，检索可以分为两步：

  1. 使用 embedding 模型对文本进行 向量化处理
  2. 计算两个向量的 cosine距离作为相似度，对相似度排序后，来选择文本。

Q8：RAG是如何使用检索到的数据进行文本生成的？

参考Part4部分，Part4介绍了如何新建一个提示词模型，创建langchain中的chain，以及如何使用chain进行检索：

  1. 创建一个prompt提示词模版，提示词模版是方便我们准备LLM需要的输入。
  2. 建立一个chain（简单chain的例子）
  3. 执行检索

注：RunnablePassthrough() 表示获取来自invoke中的文本

Q11：如何使用Huggingface的LLM模型？

方法一：

from transformers import AutoModelForCausalLM, AutoTokenizer, pipeline
model_id = "THUDM/chatglm3-6b"
tokenizer = AutoTokenizer.from_pretrained(model_id,trust_remote_code=True)
model = AutoModelForCausalLM.from_pretrained(model_id, trust_remote_code=True, device_map="auto")
pipe = pipeline(
    "text-generation", model=model, tokenizer=tokenizer, max_new_tokens=2000
)
llm = HuggingFacePipeline(pipeline=pipe)

方法二：

from langchain.llms import HuggingFacePipeline
model_id = "THUDM/chatglm3-6b"
llm = HuggingFacePipeline.from_model_id(
    model_id=model_id,
    task="text-generation",
    device=0,
    model_kwargs={"temperature": 0.01, "max_length": 2000, "trust_remote_code": True},
)

Q15：什么是Runnable protocol？如何使用？

常见用法：1 通过RunnablePassthrough读取输入

# Chain
rag_chain = (
    {"context": retriever | format_docs, "question": RunnablePassthrough()}
    | prompt
    | llm
    | StrOutputParser()
)

# Question
rag_chain.invoke("What is Task Decomposition?")

常见用法：2：

from langchain_core.prompts import ChatPromptTemplate
from langchain_openai import ChatOpenAI

model = ChatOpenAI()
prompt = ChatPromptTemplate.from_template("tell me a joke about {topic}")
chain = prompt | model
#Invoke
chain.invoke({"topic": "bears"})
#Batch：
chain.batch([{"topic": "bears"}, {"topic": "cats"}])

#Async Invoke
await chain.ainvoke({"topic": "bears"})

更多用法，请参考：https://python.langchain.com/docs/expression_language/interface/

常见用法：2

from langchain_anthropic import ChatAnthropic
from langchain_core.output_parsers import StrOutputParser
from langchain_core.prompts import ChatPromptTemplate

prompt = ChatPromptTemplate.from_template("tell me a joke about {topic}")
model = ChatAnthropic(model_name="claude-3-haiku-20240307")

chain = prompt | model | StrOutputParser()

chain.invoke({"topic": "bears"})

更多用法，请参考：https://python.langchain.com/docs/expression_language/primitives/sequence/
